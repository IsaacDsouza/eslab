Reset_Handler
    MOV R0, #0
    LDRH R1, NUM1
    LDRH R2, NUM1
    LDRH R3, NUM2
    LDR R4, =DST
    MUL R5, R2, R3
    CMP R2, R3
    BNE LOOP

LOOP
    CMP R2, R3
    BHI B1
    BLS B2

B2
    SUB R3, R2

CM
    CMP R2, R3
    BNE LOOP
    CMP R2, R5
    BLS DIVLOOP

DIVLOOP
    SUB R5, R2
    ADD R0, #1
    CMP R2, R5
    BLS DIVLOOP
